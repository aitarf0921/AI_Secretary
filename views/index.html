快速開始：嵌入程式碼

<script>
  // 替換 YOUR_OPENAI_API_KEY 為您的 OpenAI API Key
  const API_KEY = 'YOUR_OPENAI_API_KEY';
  // 新增：OpenAI Widget 初始化腳本（修正版，使用 addEventListener 避免 inline 事件污染）
  (function() {
    const widget = document.createElement('div');
    widget.id = 'openai-widget';
    widget.innerHTML = `
      <div id="widget-header">
        <span>AI 助手</span>
        <i class="fas fa-times" id="close-widget"></i>
      </div>
      <div id="widget-messages"></div>
      <div id="widget-input">
        <input id="message-input" placeholder="輸入訊息...">
        <button id="send-btn">傳送</button>
      </div>
    `;
    document.body.appendChild(widget);

    const toggle = document.createElement('div');
    toggle.id = 'widget-toggle';
    toggle.innerHTML = '<i class="fas fa-comment-dots"></i>';
    document.body.appendChild(toggle);

    // 使用 addEventListener 綁定事件，避免 inline 事件導致污染
    const closeBtn = document.getElementById('close-widget');
    closeBtn.addEventListener('click', () => widget.classList.remove('open'));
    toggle.addEventListener('click', () => widget.classList.toggle('open'));

    const messageInput = document.getElementById('message-input');
    const sendBtn = document.getElementById('send-btn');
    messageInput.addEventListener('keypress', (e) => {
      if (e.key === 'Enter') sendMessage();
    });
    sendBtn.addEventListener('click', sendMessage);

    async function sendMessage() {
      const msg = messageInput.value.trim();
      if (!msg) return;
      const messages = document.getElementById('widget-messages');
      messages.innerHTML += `<div class="message user">${msg}</div>`;
      messageInput.value = '';
      messages.scrollTop = messages.scrollHeight;

      try {
        const res = await fetch('https://api.openai.com/v1/chat/completions', {
          method: 'POST',
          headers: { 'Authorization': \`Bearer \${API_KEY}\`, 'Content-Type': 'application/json' },
          body: JSON.stringify({ model: 'gpt-3.5-turbo', messages: [{ role: 'user', content: msg }] })
        });
        const data = await res.json();
        messages.innerHTML += `<div class="message bot">${data.choices[0].message.content}</div>`;
        messages.scrollTop = messages.scrollHeight;
      } catch (e) {
        messages.innerHTML += `<div class="message bot">錯誤：\${e.message}</div>`;
      }
    }
    console.log('[OpenAI Widget] 載入成功！');
  })();
</script>

自訂您的 AI Chatbot Widget 參數，實現無縫整合

屬性 必填 說明
API_KEY 必填 OpenAI API 金鑰（從 platform.openai.com 獲取）。
model 可選 模型（如 'gpt-3.5-turbo'，預設）。
position 可選 浮動按鈕位置（預設右下）。
accent 可選 品牌主色（HEX，如 '#25d366'）。
width/height 可選 聊天面板尺寸（px，預設 360x520）。
z-index 可選 層級，確保浮在最上層。
